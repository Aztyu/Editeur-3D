179  SingleObject* base_object = this->selected_object;
180 
181  GroupObject* new_group = this->group_object_array.at(index);
182 
183  irr::core::matrix4 matr = base_object->getSceneNode()->getAbsoluteTransformation();
184 
185  const irr::core::matrix4 group_inv(new_group->getSceneNode()->getAbsoluteTransformation(), irr::core::matrix4::EM4CONST_INVERSE);
186 
187  matr = (group_inv*matr);
188 
189  new_group->addMember(base_object);
190 
191  base_object->getSceneNode()->setPosition(matr.getTranslation());
192  base_object->getSceneNode()->setRotation(matr.getRotationDegrees());
193 
194  base_object->getSceneNode()->updateAbsolutePosition();
195  irr::core::vector3df scale_base = new_group->getScale();
196  irr::core::vector3df scale_mod = scale_base.invert();

199  }else{
